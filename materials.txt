create database materialHousedb;
create table users(
	id serial primary key,
	username varchar(100) unique not null,
	password text not null,
	role varchar(20) not null check (role IN ('admin', 'storekeeper', 'accountant')),
	name varchar(100),
	secondname varchar(100),
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	email VARCHAR(100),
	phone VARCHAR(20),
	birthday DATE,
	updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
)

create table notifications (
	id serial primary key,
	user_id int references users(id) on delete set null,
	type varchar(100) not null,
	title varchar(255) not null,
	message text not null,
	read boolean default false,
	created_at timestamp default current_timestamp
);


create table materialCategories (
	id serial primary key,
	name varchar(255) not null,
	description text,
	created_by int references users(id) on delete set null,
	updated_by int references users(id) on delete set null,
	created_at timestamp default current_timestamp,
	updated_at timestamp default current_timestamp
 );
create table materials (
	id serial primary key,
	category_id int references materialCategories(id) on delete set null,
	name varchar(255) not null,
	code varchar(50) unique not null,
	description text,
	unit varchar(30) not null,
	quantity int default 0 check (quantity >=0),
	created_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    updated_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
 );
CREATE TABLE backups (
    id SERIAL PRIMARY KEY,
    filename VARCHAR(255) NOT NULL,
    filepath VARCHAR(500) NOT NULL,
    file_size BIGINT,
    created_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    description TEXT
);
create table material_requests (
	id serial primary key,
	request_type varchar(20) not null check (request_type in ('incoming', 'outgoing')),
	status varchar(20) not null default 'pending' check (status in ('pending', 'approved', 'rejected')),
	created_by int references users(id) on delete set null,
	reviewed_by int references users(id) on delete set null,
	notes text,
	rejection_reason text,
	created_at timestamp default current_timestamp,
	reviewed_at timestamp,
	is_public BOOLEAN DEFAULT TRUE,
	title VARCHAR(255)
)
CREATE TABLE material_requests_items (
    id SERIAL PRIMARY KEY,
    request_id INTEGER NOT NULL REFERENCES material_requests(id) ON DELETE CASCADE,
    material_id INTEGER NOT NULL REFERENCES materials(id) ON DELETE CASCADE,
    quantity INTEGER NOT NULL CHECK (quantity > 0),
    current_quantity_at_request INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
)




--INVENTORY

-- Таблица инвентаризаций
CREATE TABLE inventories (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    status VARCHAR(20) DEFAULT 'draft',
    created_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    responsible_person INTEGER REFERENCES users(id) ON DELETE SET NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP,
    approved_at TIMESTAMP,
    approved_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    cancelled_at TIMESTAMP,
    cancelled_by INTEGER REFERENCES users(id) ON DELETE SET NULL,
    CONSTRAINT inventories_status_check CHECK (status IN ('draft', 'in_progress', 'completed', 'approved', 'cancelled', 'expired'))
);

-- Категории в инвентаризации
CREATE TABLE inventory_categories (
    id SERIAL PRIMARY KEY,
    inventory_id INTEGER REFERENCES inventories(id) ON DELETE CASCADE,
    category_id INTEGER REFERENCES materialcategories(id) ON DELETE CASCADE
);

-- Конкретные товары в инвентаризации
CREATE TABLE inventory_materials (
    id SERIAL PRIMARY KEY,
    inventory_id INTEGER REFERENCES inventories(id) ON DELETE CASCADE,
    material_id INTEGER REFERENCES materials(id) ON DELETE CASCADE
);

-- Результаты инвентаризации
CREATE TABLE inventory_results (
    id SERIAL PRIMARY KEY,
    inventory_id INTEGER REFERENCES inventories(id) ON DELETE CASCADE,
    material_id INTEGER REFERENCES materials(id) ON DELETE CASCADE,
    system_quantity INTEGER NOT NULL,
    actual_quantity INTEGER,
    difference INTEGER GENERATED ALWAYS AS (actual_quantity - system_quantity) STORED,
    reason TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- чаты

CREATE TABLE IF NOT EXISTS chats (
    id SERIAL PRIMARY KEY,
    user1_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    user2_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_by_user1 BOOLEAN DEFAULT FALSE,
    deleted_by_user2 BOOLEAN DEFAULT FALSE,
    UNIQUE(user1_id, user2_id)
);


CREATE TABLE IF NOT EXISTS messages (
    id SERIAL PRIMARY KEY,
    chat_id INTEGER REFERENCES chats(id) ON DELETE CASCADE,
    sender_id INTEGER REFERENCES users(id) ON DELETE SET NULL,
    message TEXT,
    image_url TEXT,
    file_url TEXT,
    file_name TEXT,
    file_size BIGINT,
    is_read BOOLEAN DEFAULT FALSE,
    read_at TIMESTAMP,
    is_deleted_for_sender BOOLEAN DEFAULT FALSE,
    is_deleted_for_receiver BOOLEAN DEFAULT FALSE,
    edited_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX IF NOT EXISTS idx_messages_chat_id ON messages(chat_id);
CREATE INDEX IF NOT EXISTS idx_messages_created_at ON messages(created_at);
CREATE INDEX IF NOT EXISTS idx_chats_users ON chats(user1_id, user2_id);
CREATE INDEX IF NOT EXISTS idx_chats_updated_at ON chats(updated_at);
